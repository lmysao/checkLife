   505→  const row = { label, done:false, position, day_key: todayKey(), priority: priority||"media" };

   509→    cache.tasks.push(data);
   510→  }catch(e){

   517→async function toggleTask(id){

   522→  try{ await sb.from("tasks").update({done:t.done}).eq("id", id); }catch(e){}

   524→async function deleteTask(id){
   525→  cache.tasks = cache.tasks.filter(t=>t.id!==id);

   529→async function moveTask(id, dir){
   530→  const list = cache.tasks.filter(t=>t.day_key===todayKey()).sort((a,b)=>a.position-b.position);
   531→  const idx = list.findIndex(t=>t.id===id);
   532→  const swapIdx = idx+dir;
   533→  if(swapIdx<0 || swapIdx>=list.length) return;
   534→  const a=list[idx], b=list[swapIdx];
   535→  const tmp=a.position; a.position=b.position; b.position=tmp;
   536→  saveCache(); render();
   537→  try{
   538→    await sb.from("tasks").update({position:a.position}).eq("id", a.id);
   539→    await sb.from("tasks").update({position:b.position}).eq("id", b.id);
   540→  }catch(e){}
   541→}

   820→function lastNDays(n){
   821→  const arr = [];
   822→  for(let i=n-1; i>=0; i--){
   823→    const d = new Date();
   824→    d.setDate(d.getDate()-i);
   825→    arr.push(d.getFullYear()+"-"+String(d.getMonth()+1).padStart(2,"0")+"-"+String(d.getDate()).padStart(2,"0"));
   826→  }
   827→  return arr;
   828→}
   829→function computeCompletionPct(moduleKey, n){
   830→  const items = cache.items.filter(i=>i.module===moduleKey && i.active);
   831→  if(items.length===0) return 0;
   832→  const days = lastNDays(n);
   833→  let totalPossible=0, totalDone=0;
   834→  days.forEach(day=>{
   835→    items.forEach(it=>{
   836→      const st = cache.status.find(s=>s.item_id===it.id && s.period_key===day);
   837→      if(st){ totalPossible++; if(st.checked) totalDone++; }
   838→    });
   839→  });
   840→  if(totalPossible===0) return 0;
   841→  return Math.round(totalDone/totalPossible*100);
   842→}
   843→function computeStreak(moduleKey){
   844→  const items = cache.items.filter(i=>i.module===moduleKey && i.active);
   845→  if(items.length===0) return 0;
   846→  let streak = 0;
   847→  for(let i=0;i<365;i++){
   848→    const d = new Date();
   849→    d.setDate(d.getDate()-i);
   850→    const key = d.getFullYear()+"-"+String(d.getMonth()+1).padStart(2,"0")+"-"+String(d.getDate()).padStart(2,"0");
   851→    const allChecked = items.every(it=>{
   852→      const st = cache.status.find(s=>s.item_id===it.id && s.period_key===key);
   853→      return st && st.checked;
   854→    });
   855→    if(allChecked) streak++;
   856→    else break;
   857→  }
   858→  return streak;
   859→}
