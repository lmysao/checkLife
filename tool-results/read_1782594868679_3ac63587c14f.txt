   820→   820→function lastNDays(n){

   822→   822→  for(let i=n-1; i>=0; i--){
   823→   823→    const d = new Date();
   824→   824→    d.setDate(d.getDate()-i);
   825→   825→    arr.push(d.getFullYear()+"-"+String(d.getMonth()+1).padStart(2,"0")+"-"+String(d.getDate()).padStart(2,"0"));
   826→   826→  }
   827→   827→  return arr;

   829→   829→function computeCompletionPct(moduleKey, n){
   830→   830→  const items = cache.items.filter(i=>i.module===moduleKey && i.active);
   831→   831→  if(items.length===0) return 0;
   832→   832→  const days = lastNDays(n);
   833→   833→  let totalPossible=0, totalDone=0;
   834→   834→  days.forEach(day=>{
   835→   835→    items.forEach(it=>{
   836→   836→      const st = cache.status.find(s=>s.item_id===it.id && s.period_key===day);
   837→   837→      if(st){ totalPossible++; if(st.checked) totalDone++; }
   838→   838→    });
   839→   839→  });
   840→   840→  if(totalPossible===0) return 0;
   841→   841→  return Math.round(totalDone/totalPossible*100);
   842→   842→}
   843→   843→function computeStreak(moduleKey){
   844→   844→  const items = cache.items.filter(i=>i.module===moduleKey && i.active);
   845→   845→  if(items.length===0) return 0;
   846→   846→  let streak = 0;
   847→   847→  for(let i=0;i<365;i++){
   848→   848→    const d = new Date();
   849→   849→    d.setDate(d.getDate()-i);
   850→   850→    const key = d.getFullYear()+"-"+String(d.getMonth()+1).padStart(2,"0")+"-"+String(d.getDate()).padStart(2,"0");
   851→   851→    const allChecked = items.every(it=>{
   852→   852→      const st = cache.status.find(s=>s.item_id===it.id && s.period_key===key);
   853→   853→      return st && st.checked;
   854→   854→    });
   855→   855→    if(allChecked) streak++;
   856→   856→    else break;
   857→   857→  }
   858→   858→  return streak;
   859→   859→}
